// ТИТРЫ ДЛЯ HALF-LIFE
// Команда позиционирования $position x y
// x и y — значения от 0 до 1 для независимости от разрешения экрана
// -1 означает центр в каждом измерении
// Команда эффекта $effect <номер эффекта>
// эффект 0 — появление/затухание (fade in/fade out)
// эффект 1 — мерцающие титры (flickery credits)
// эффект 2 — побуквенное появление текста (как в тренировочном режиме)
// Команда цвета текста r g b — $color
// время появления, время исчезновения / время удержания на экране
// $fadein — время появления текста (для эффекта 2 — на каждый символ)
// $fadeout — время затухания текста
// $holdtime — время отображения текста на экране

// ВСТУПИТЕЛЬНЫЕ ТИТРЫ
// Пока всё по центру
$position -1 -1
$effect 2
// Это основной цвет
$color 100 100 100
// Это цвет выделения
$color2 240 110 0
// Время появления каждого символа
$fadein 0.01
// Время задержки цвета выделения относительно ведущего края текста
$fxtime 0.25
// Время удержания сообщения на экране (после полного появления)
$holdtime 3.5
// Время затухания сообщения после удержания
$fadeout 1.5
